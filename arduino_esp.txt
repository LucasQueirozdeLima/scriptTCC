#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN 10
#define RST_PIN 9

MFRC522 mfrc522(SS_PIN, RST_PIN); // Instância do MFRC522

void setup() {
  Serial.begin(115200);  // Inicia comunicação com o monitor serial
  SPI.begin();           // Inicia o barramento SPI
  mfrc522.PCD_Init();    // Inicia o MFRC522
  
  Serial.println("Aproxime a tag do leitor...");
}

void loop() {
  // Verifica se há uma tag presente
  if (mfrc522.PICC_IsNewCardPresent()) {
    Serial.println("Tag detectada!");

    // Lê o ID da tag
    if (mfrc522.PICC_ReadCardSerial()) {
      Serial.print("ID da tag: ");
      for (byte i = 0; i < mfrc522.uid.size; i++) {
        // Imprime o ID da tag em formato hexadecimal
        Serial.print(mfrc522.uid.uidByte[i], HEX);
      }
      Serial.println();
      mfrc522.PICC_HaltA();  // Interrompe a comunicação com a tag
    }
  }
  delay(1000);  // Aguarda 1 segundo antes de tentar ler novamente
}
